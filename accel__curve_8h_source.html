91  void reset(const float j_max, const float a_max, const float v_start,
92  const float v_end) {
93  /* 符号付きで代入 */
94  am = (v_end > v_start) ? a_max : -a_max; //< 最大加速度の符号を決定
95  jm = (v_end > v_start) ? j_max : -j_max; //< 最大躍度の符号を決定
96  /* 初期値と最終値を代入 */
97  v0 = v_start; //< 代入
98  v3 = v_end; //< 代入
99  t0 = 0; //< ここでは初期値をゼロとする
100  x0 = 0; //< ここでは初期値はゼロとする
101  /* 速度が曲線となる部分の時間を決定 */
102  const auto tc = a_max / j_max;
103  /* 等加速度直線運動の時間を決定 */
104  const auto tm = (v3 - v0) / am - tc;
105  /* 等加速度直線運動の有無で分岐 */
106  if (tm > 0) {
107  /* 速度: 曲線 -> 直線 -> 曲線 */
108  t1 = t0 + tc;
109  t2 = t1 + tm;
110  t3 = t2 + tc;
111  v1 = v0 + am * tc / 2; //< v(t) を積分
112  v2 = v1 + am * tm; //< v(t) を積分
113  x1 = x0 + v0 * tc + am * tc * tc / 6; //< x(t) を積分
114  x2 = x1 + v1 * tm; //< x(t) を積分

116  } else {

122  x1 = x2 = x0 + v1 * tcp + jm * tcp * tcp * tcp / 6; //< x(t) を積分

244  static float calcReachableVelocityEnd(const float j_max, const float a_max,
245  const float vs, const float vt,
246  const float d) {
247  /* 速度が曲線となる部分の時間を決定 */
248  const auto tc = a_max / j_max;
249  /* 最大加速度の符号を決定 */
250  const auto am = (vt > vs) ? a_max : -a_max;
251  const auto jm = (vt > vs) ? j_max : -j_max;
252  /* 等加速度直線運動の有無で分岐 */
253  const auto d_triangle = (vs + am * tc / 2) * tc; //< distance @ tm == 0
254  const auto v_triangle = jm / am * d - vs; //< v_end @ tm == 0
255  // logd << "d_tri: " << d_triangle << std::endl;
256  // logd << "v_tri: " << v_triangle << std::endl;
257  if (d * v_triangle > 0 && std::abs(d) > std::abs(d_triangle)) {
258  /* 曲線・直線・曲線 */
259  logd << "v: curve - straight - curve" << std::endl;
260  /* 2次方程式の解の公式を解く */
261  const auto amtc = am * tc;
262  const auto D = amtc * amtc - 4 * (amtc * vs - vs * vs - 2 * am * d);
263  const auto sqrtD = std::sqrt(D);
264  return (-amtc + (d > 0 ? sqrtD : -sqrtD)) / 2;
265  }
266  /* 曲線・曲線 (走行距離が短すぎる) */
267  /* 3次方程式を解いて，終点速度を算出;
268  * 簡単のため，値を一度すべて正に変換して，計算結果に符号を付与して返送 */
269  const auto a = std::abs(vs);
270  const auto b = (d > 0 ? 1 : -1) * jm * d * d;
271  const auto aaa_27 = a * a * a / 27;
272  const auto cr = 8 * aaa_27 + b / 2;
273  const auto ci_b = 8 * aaa_27 / b + float(1) / 4;
274  if (ci_b >= 0) {
275  /* ルートの中が非負のとき */
276  logd << "v: curve - curve (accel)" << std::endl;
277  const auto cc = std::cbrt(cr + std::abs(b) * std::sqrt(ci_b));
278  return (d > 0 ? 1 : -1) *
279  (cc + 4 * a * a / cc / 9 - a / 3); //< 3次方程式の解
280  } else {
281  /* ルートの中が負のとき */
282  logd << "v: curve - curve (decel)" << std::endl;
283  const auto cc =
284  std::pow(std::complex<float>(cr, std::abs(b) * std::sqrt(-ci_b)),
285  float(1) / 3);
286  return (d > 0 ? 1 : -1) * (cc.real() * 2 - a / 3); //< 3次方程式の解
287  }

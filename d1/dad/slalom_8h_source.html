128  static void integrate(const AccelDesigner &ad, State &s, const float v,
129  const float t, const float Ts, const float k_slip = 0) {
130  /* Calculation */
131  const std::array<float, 3> th{{ad.x(t), ad.x(t + Ts / 2), ad.x(t + Ts)}};
132  const std::array<float, 3> w{{ad.v(t), ad.v(t + Ts / 2), ad.v(t + Ts)}};
133  std::array<float, 3> cos_th;
134  std::array<float, 3> sin_th;
135  for (int i = 0; i < 3; ++i) {
136  const auto th_slip = std::atan(-k_slip * v * w[i]);
137  cos_th[i] = std::cos(th[i] + th_slip);
138  sin_th[i] = std::sin(th[i] + th_slip);
139  }
140  /* Runge-Kutta Integral */
141  s.q.x += v * Ts * (cos_th[0] + 4 * cos_th[1] + cos_th[2]) / 6;
142  s.q.y += v * Ts * (sin_th[0] + 4 * sin_th[1] + sin_th[2]) / 6;
143  /* Result */
144  s.dq.x = v * cos_th[2];
145  s.dq.y = v * sin_th[2];
146  s.q.th = ad.x(t + Ts);
147  s.dq.th = ad.v(t + Ts);
148  s.ddq.th = ad.a(t + Ts);
149  s.dddq.th = ad.j(t + Ts);
150  s.ddq.x = -s.dq.y * s.dq.th;
151  s.ddq.y = +s.dq.x * s.dq.th;
152  s.dddq.x = -s.ddq.y * s.dq.th - s.dq.y * s.ddq.th;
153  s.dddq.y = +s.ddq.x * s.dq.th + s.dq.x * s.ddq.th;
154  }
